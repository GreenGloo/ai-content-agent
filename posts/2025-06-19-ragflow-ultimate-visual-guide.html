🎯 See RAGFlow in Action

# Quick Start with RAGFlow
pip install ragflow

from ragflow import RAGEngine

# Initialize the engine
engine = RAGEngine(
    model="claude-3-sonnet",
    chunk_size=512,
    overlap=50
)

# Process your documents
documents = engine.load_documents("./your_docs/")
index = engine.create_index(documents)

# Ask questions about your documents
response = engine.query(
    "What are the key findings in the Q3 report?",
    index=index
)

print(response.answer)
# Output: "The Q3 report shows 23% revenue growth..."
